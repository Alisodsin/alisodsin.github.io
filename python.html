        n = n+1

    if re.findall("\s+", x[i]):
        print(f"{n}-{x[i]}")
        n = n+1


# --?-- i.e 0 or 1

    if re.findall("\s?", x[i]):
        print(f"{n}-{x[i]}")
        n = n+1

    if re.findall("\s{1}", x[i]):
        print(f"{n}-{x[i]}")
        n = n+1

for i in range(len(x)):
    if re.findall("c{2,5}", x[i]):
        print(f"{n}-{x[i]}")
        n = n+1


# --{a,}-- i.e a or more

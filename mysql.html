the REGEXP operator

--contain e perceded by a letter range[a-h] '[a-h]e'

    orders.order_id

--here we insert all customers records wheather they have orders or NOT [LEFT JOIN]

    customers.customer_id,
    customers.first_name,
    orders.order_id
FROM customers
    LEFT JOIN orders on customers.customer_id = orders.customer_id

    customers.customer_id;

--insert all the orders record (RIGHT) but this is the same as first

    orders.order_id

    RIGHT JOIN orders on customers.customer_id = orders.customer_id

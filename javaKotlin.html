    }

import java.util.Scanner;

    // while loop = executes a block of code as long as  it's condition remains true
    public static void main(String[] args) {
        // while
        Scanner scanner = new Scanner(System.in);
        String name = "";
        while (name.isBlank()) {
            System.out.print("Enter your name: ");
            name = scanner.nextLine();
        }
        System.out.println("Welcome " + name);
        //do while
        int age = 0;
        do {
            System.out.println("Enter your age");
            age = scanner.nextInt();
        }
        while (!(age >= 0 && age <= 100));
        System.out.println("you are " + age + " years old");
    }
}

    // wrapper class =  provides a way to use primitive data types as reference data types
    // reference data types contain useful methods
    // can be used with collections (ex.ArrayList)
    //primitive  //wrapper
    //---------  //-------
    // boolean  Boolean
    // char   Character
    // int   Integer
    // double  Double
    // autoboxing = the automatic conversion that the Java compiler makes between the primitive types and their corresponding object wrapper classes
    // unboxing = the reverse of autoboxing. Automatic conversion of wrapper class to primitive
    public static void main(String[] args) {
        // Autoboxing
        int a = 20;
        Integer b = Integer.valueOf(a); // boxing
        Integer c = a; //Autoboxing => convert primitive to objects
        System.out.println(a + " " + b + " " + c);

        // Unboxing
        Integer d = 3;
        int e = d.intValue(); //converting Integer to int explicitly
        int f = d; //unboxing, now compiler will write a.intValue() internally

    }
}

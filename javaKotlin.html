Java

import java.util.Scanner;

// switch = an efficient alternative to many else if statements
public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("what day is today? ");
        String day = scanner.nextLine();
        //switch statement
        switch (day) {
            case "Monday" -> System.out.println("It's Monday!");
            case "Tuesday" -> System.out.println("It's Tuesday!");
            case "Wednesday" -> System.out.println("It's Wednesday!");
            case "Thursday" -> System.out.println("It's Thursday!");
            case "Friday" -> System.out.println("It's Friday!");
            case "Saturday" -> System.out.println("It's Saturday!");
            case "Sunday" -> System.out.println("It's Sunday!");
            default -> System.out.println(day + " is not a day!");
        }
    }
}

import java.util.Scanner;

public class Main {
    // while loop = executes a block of code as long as  it's condition remains true
    public static void main(String[] args) {
        // while
        Scanner scanner = new Scanner(System.in);
        String name = "";
        while (name.isBlank()) {
            System.out.print("Enter your name: ");
            name = scanner.nextLine();
        }
        System.out.println("Welcome " + name);
        //do while
        int age = 0;
        do {
            System.out.println("Enter your age");
            age = scanner.nextInt();
        }
        while (!(age >= 0 && age <= 100));
        System.out.println("you are " + age + " years old");
    }
}

import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        // nested loops = a loop inside a loop
        System.out.println("How many rows");
        int rows = scanner.nextInt();
        System.out.println("How many columns?:");
        int columns = scanner.nextInt();
        scanner.nextLine(); // clear buffer for the next input
        System.out.println("what symbol");
        String symbol = scanner.nextLine();
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < columns; j++) {
                System.out.print(symbol);
            }
            System.out.println();
        }
    }
}

public class Main {
    // wrapper class =  provides a way to use primitive data types as reference data types
    // reference data types contain useful methods
    // can be used with collections (ex.ArrayList)
    //primitive  //wrapper
    //---------  //-------
    // boolean  Boolean
    // char   Character
    // int   Integer
    // double  Double
    // autoboxing = the automatic conversion that the Java compiler makes between the primitive types and their corresponding object wrapper classes
    // unboxing = the reverse of autoboxing. Automatic conversion of wrapper class to primitive
    public static void main(String[] args) {
        // Autoboxing
        int a = 20;
        Integer b = Integer.valueOf(a); // boxing
        Integer c = a; //Autoboxing => convert primitive to objects
        System.out.println(a + " " + b + " " + c);

        // Unboxing
        Integer d = 3;
        int e = d.intValue(); //converting Integer to int explicitly
        int f = d; //unboxing, now compiler will write a.intValue() internally
        System.out.println(d + " " + e + " " + f);
    }
}

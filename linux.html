cp

cp file1 file2      #Copy file1 to file2. If file2 exists, it is overwritten with the contents of file1. If file2 does not exist, it is created.
cp -i file1 file2   #Same as previous command, except that if file2 exists, the user is prompted before it is overwritten.
cp file1 file2 dir1 #Copy file1 and file2 into directory dir1. The directory dir1 must already exist.
cp dir1/* dir2      #Using a wildcard, copy all the files in dir1 into dir2. The directory dir2 must already exist.
cp -r dir1 dir2     #Copy the contents of directory dir1 to directory dir2. If directory dir2 does not exist, it is created and, after the copy, will contain the same contents as directory dir1. If directory dir2 does exist, then directory dir1 (and its contents) will be copied into dir2.
cp -u file1 file2   #copy only when the file1 is newer than the file2 or when the file2 is missing

cut

#!/bin/bash
cut -b 1,2,3 state.txt      # cut the first,second and the 3rd byte for each line and output it
cut -b 1-3,5-7 state.txt    # cut form 1 to 3rd byte plus from 5th to 7th bytes
cut -b 1- state.txt         # select the all line and output it
cut -c 2,5,7 state.txt      # cut the 2nd+5th+7th character from each line and output it
cut -c 1-7 state.txt        # List with ranges
cut -c 1- state.txt         # list contain (1-) select all all line characters
cut -d " " -f 1 state.txt   # cut field 1 (-f 1) when the fields delimiter is space (-d " ")
cut -d " " -f 1-2 state.txt # cut the 1st and the 2nd field when the fields delimiter is space (-d " ")
cut --complement -d " " -f 2 state.txt             # (–complement) As the name suggests it complement the output
cut -d " " -f 1,2 state.txt --output-delimiter="%" # –output-delimiter

Recursion

c#
public class Program
{
    public static int x = 1;
    public static void Main()
    {
        rtr();
    }
    public static void rtr()
    {
        int z = x;
        Console.WriteLine($"i'm function mumber {z}");
        x++;
        while (x <= 20)
        {
            rtr();
        }
        Console.WriteLine($"i'm function mumber {z}");
    }
}
powershell
$global:x = 1
function rtr() {
    $z = $global:x
    "i'm function mumber $z"
    $global:x++
    while ( $global:x -le 20 ) {
        rtr
    }
    "i'm function mumber $z"
}
rtr

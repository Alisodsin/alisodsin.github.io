        Thread thread2 = new(()=>countdown());

        thread2.Start();

        for (int i = 0; i <= 5; i++)
        {
            Console.WriteLine (i);
            if (i == 5)
            {
                Console.WriteLine("counting end");
               
            }

        }
    }

    {
        for (int i = 5; i >= 0; i--)
        {
            Console.WriteLine (i);
            if (i == 0)
            {
                Console.WriteLine("counting end");
            }
            Thread.Sleep(1000);
        }
    }
}

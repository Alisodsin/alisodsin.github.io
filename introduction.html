List Comprehension

List comprehension offers a shorter syntax when you want to create a new list based on the values of an existing list.

fruits = ["apple", "banana", "cherry", "kiwi", "mango"]
newlist1 = [x for x in fruits if "a" in x]
print(newlist1) #return ['apple', 'banana', 'mango']
newlist2 = [x for x in range(10) if x < 5]
print(newlist2) # output [0, 1, 2, 3, 4]

fruits = ["apple", "banana", "cherry", "kiwi", "mango"]
newlist = [x.upper() for x in fruits]
print(newlist) # ['APPLE', 'BANANA', 'CHERRY', 'KIWI', 'MANGO']

fruits = ["apple", "banana", "cherry", "kiwi", "mango"]
newlist = [x if x != "banana" else "orange" for x in fruits]
print(newlist) # ['apple', 'orange', 'cherry', 'kiwi', 'mango']

inheritance

# father class
class Person:
    def __init__(self, name, age):
        self.name = name
        self.age = age

    def printname(self):
        print(self.name, self.age)


# son class (inheritor)
# 1-the son will be the same as father
class Student(Person):
    pass


# 2-the son won't be the same as father
class Worker(Person):
    def __init__(self, name, age, tall):
        super().__init__(name, age)
        self.tall = tall  # here son(Worker) has extrea property

    def printname(self):
        super().printname()
        print(self.tall)  # modify the inherited printname method


# instantization
a = Person("ali", 25)
b = Worker("soso", 35, 170)
a.printname()  # output ali 25
b.printname()  # output soso 35 and then 170

If you add a method in the child class with the same name as a function in the parent class, the inheritance of the parent method will be overridden.

# example
import re
file = open("american-english", "r")
x = file.read().split("\n")
file.close()
for i in range(len(x)):
    if re.findall("^[et]a", x[i]):  # match any string begins with "ea" or "ta"
        print(f"{i+1} {x[i]}")

2-\
Signals a special sequence (can also be used to escape special characters)

# \A - Returns a match if the specified characters are at the beginning of the string
import re
file = open("american-english", "r")
x = file.read().split("\n")
file.close()
for i in range(len(x)):
    if re.findall("\Aon", x[i]):  # match any string begins with "on"
        print(f"{i} {x[i]}")

# \b - Returns a match where the specified characters are at the beginning or at the end of a word (the "r" in the beginning is making sure that the string is being treated as a "raw string")
import re
file = open("american-english", "r")
x = file.read().split("\n")
file.close()
for i in range(len(x)):
    # match any word begins or ends with "on","r" for raw string
    if re.findall(r"\bon", x[i]):
        print(f"{i} {x[i]}")

# \B Returns a match where the specified characters are present, but NOT at the beginning (or at the end) of a word
import re
file = open("american-english", "r")
x = file.read().split("\n")
file.close()
for i in range(len(x)):
    # match any word contains "on" but not begin or end with it
    if re.findall(r"\Bon", x[i]):
        print(f"{i} {x[i]}")

# \d - Returns a match where the string contains digits (numbers from 0-9)
import re
file = open("american-english", "r")
x = file.read().split("\n")
file.close()
for i in range(len(x)):
    if re.findall(r"\d", x[i]):  # match any word contains digits [0-9]
        print(f"{i} {x[i]}")


# \D Returns a match where the string DOES NOT contain digits
import re
file = open("american-english", "r")
x = file.read().split("\n")
file.close()
for i in range(len(x)):
    if re.findall(r"\D", x[i]):  # match any word NOT contains digits [0-9]
        print(f"{i} {x[i]}")

# \s Returns a match where the string contains a white space character
import re
file = open("american-english", "r")
x = file.read().split("\n")
file.close()

Sets

  1. [arn]         Returns a match where one of the specified characters (a, r, or n) are present
  2. [a-n]         Returns a match for any lower case character, alphabetically between a and n
  3. [^arn]       Returns a match for any character EXCEPT a, r, and n
  4. [0123]       Returns a match where any of the specified digits (0, 1, 2, or 3) are present
  5. [0-9]         Returns a match for any digit between 0 and 9
  6. [0-5][0-9] Returns a match for any two-digit numbers from 00 and 59
  7. [a-zA-Z]    Returns a match for any character alphabetically between a and z, lower case OR upper case
  8. [+]           In sets, +, *, ., |, (), $,{} has no special meaning, so [+] means: return a match for any + character in the string
